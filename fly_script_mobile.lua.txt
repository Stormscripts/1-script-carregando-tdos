-- Script Local de Fly (Voo) - Versão Mobile
-- Uso: Local script para mecânica de voo em celular

local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local TouchInputService = game:GetService("UserInputService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoidRootPart = character:WaitForChild("HumanoidRootPart")
local playerGui = player:WaitForChild("PlayerGui")

-- Configurações
local flySpeed = 50
local isFlying = false
local flying = nil

-- Variáveis de toque
local touchStartPos = nil
local touchConnection = nil
local touchMoveConnection = nil

-- Criar interface de controle
local function createMobileUI()
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "FlyUI"
    screenGui.ResetOnSpawn = false
    screenGui.Parent = playerGui
    
    -- Botão toggle Fly
    local flyButton = Instance.new("TextButton")
    flyButton.Name = "FlyButton"
    flyButton.Size = UDim2.new(0, 80, 0, 50)
    flyButton.Position = UDim2.new(1, -100, 0, 20)
    flyButton.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
    flyButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    flyButton.TextSize = 14
    flyButton.Font = Enum.Font.GothamBold
    flyButton.Text = "FLY ON"
    flyButton.Parent = screenGui
    
    -- Joystick direção
    local joystickBg = Instance.new("Frame")
    joystickBg.Name = "JoystickBg"
    joystickBg.Size = UDim2.new(0, 120, 0, 120)
    joystickBg.Position = UDim2.new(0, 20, 1, -140)
    joystickBg.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
    joystickBg.BackgroundTransparency = 0.5
    joystickBg.BorderRadius = UDim.new(1, 0)
    joystickBg.Parent = screenGui
    
    local joystickKnob = Instance.new("Frame")
    joystickKnob.Name = "JoystickKnob"
    joystickKnob.Size = UDim2.new(0, 50, 0, 50)
    joystickKnob.Position = UDim2.new(0.5, -25, 0.5, -25)
    joystickKnob.BackgroundColor3 = Color3.fromRGB(0, 200, 255)
    joystickKnob.BorderRadius = UDim.new(1, 0)
    joystickKnob.Parent = joystickBg
    
    -- Botões verticais (subir/descer)
    local upButton = Instance.new("TextButton")
    upButton.Name = "UpButton"
    upButton.Size = UDim2.new(0, 60, 0, 50)
    upButton.Position = UDim2.new(1, -80, 0, 20)
    upButton.BackgroundColor3 = Color3.fromRGB(0, 200, 100)
    upButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    upButton.TextSize = 16
    upButton.Font = Enum.Font.GothamBold
    upButton.Text = "⬆"
    upButton.Parent = screenGui
    
    local downButton = Instance.new("TextButton")
    downButton.Name = "DownButton"
    downButton.Size = UDim2.new(0, 60, 0, 50)
    downButton.Position = UDim2.new(1, -80, 0, 80)
    downButton.BackgroundColor3 = Color3.fromRGB(200, 100, 0)
    downButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    downButton.TextSize = 16
    downButton.Font = Enum.Font.GothamBold
    downButton.Text = "⬇"
    downButton.Parent = screenGui
    
    return {
        screenGui = screenGui,
        flyButton = flyButton,
        joystickBg = joystickBg,
        joystickKnob = joystickKnob,
        upButton = upButton,
        downButton = downButton
    }
end

-- Variáveis de controle
local keys = {
    up = false,
    down = false,
    left = false,
    right = false,
    jump = false,
    ctrl = false
}

-- Criar BodyVelocity para o voo
local function startFlying()
    if isFlying then return end
    isFlying = true
    
    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.Velocity = Vector3.new(0, 0, 0)
    bodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bodyVelocity.Parent = humanoidRootPart
    
    flying = bodyVelocity
    
    -- Loop do voo
    local connection
    connection = RunService.RenderStepped:Connect(function()
        if not isFlying or not flying or not flying.Parent then
            connection:Disconnect()
            if flying then flying:Destroy() end
            isFlying = false
            return
        end
        
        local camera = workspace.CurrentCamera
        local moveDirection = Vector3.new(0, 0, 0)
        
        -- Controles do joystick
        if keys.up or keys.jump then
            moveDirection = moveDirection + Vector3.new(0, 1, 0)
        end
        if keys.down or keys.ctrl then
            moveDirection = moveDirection - Vector3.new(0, 1, 0)
        end
        if keys.right then
            moveDirection = moveDirection + camera.CFrame.RightVector
        end
        if keys.left then
            moveDirection = moveDirection - camera.CFrame.RightVector
        end
        
        -- Normalizar e aplicar velocidade
        if moveDirection.Magnitude > 0 then
            moveDirection = moveDirection.Unit
        end
        
        flying.Velocity = moveDirection * flySpeed
    end)
end

-- Parar voo
local function stopFlying()
    if not isFlying then return end
    isFlying = false
    
    if flying then
        flying:Destroy()
        flying = nil
    end
    
    keys = {up = false, down = false, left = false, right = false, jump = false, ctrl = false}
end

-- Criar UI
local ui = createMobileUI()

-- Botão Fly
ui.flyButton.Activated:Connect(function()
    if isFlying then
        stopFlying()
        ui.flyButton.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
        ui.flyButton.Text = "FLY ON"
    else
        startFlying()
        ui.flyButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
        ui.flyButton.Text = "FLY OFF"
    end
end)

-- Botão Subir
local upPressed = false
ui.upButton.InputBegan:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch then
        upPressed = true
        keys.jump = true
    end
end)

ui.upButton.InputEnded:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch then
        upPressed = false
        keys.jump = false
    end
end)

-- Botão Descer
local downPressed = false
ui.downButton.InputBegan:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch then
        downPressed = true
        keys.ctrl = true
    end
end)

ui.downButton.InputEnded:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch then
        downPressed = false
        keys.ctrl = false
    end
end)

-- Joystick para movimento
ui.joystickBg.InputBegan:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch then
        touchStartPos = input.Position
    end
end)

ui.joystickBg.InputChanged:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch and touchStartPos then
        local delta = input.Position - touchStartPos
        local magnitude = delta.Magnitude
        
        -- Atualizar posição do knob
        if magnitude > 0 then
            local direction = delta.Unit
            local offset = math.min(magnitude, 35)
            ui.joystickKnob.Position = UDim2.new(0.5, -25 + (direction.X * offset), 0.5, -25 + (direction.Y * offset))
        end
        
        -- Detectar direções
        if math.abs(delta.X) > math.abs(delta.Y) then
            if delta.X > 30 then
                keys.right = true
                keys.left = false
            elseif delta.X < -30 then
                keys.left = true
                keys.right = false
            else
                keys.right = false
                keys.left = false
            end
        else
            keys.right = false
            keys.left = false
        end
    end
end)

ui.joystickBg.InputEnded:Connect(function(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.Touch then
        touchStartPos = nil
        keys.left = false
        keys.right = false
        ui.joystickKnob.Position = UDim2.new(0.5, -25, 0.5, -25)
    end
end)

print("Script de Fly para Mobile carregado!")
print("Clique no botão FLY ON para começar")
